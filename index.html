        {
            let A = [2];
            let evensum = 0;
            let oddsum = 0;
            for(let i=0; i < N; i++){
              if(Arr[i] % 2 === 0){
                evensum += Arr[i];
              }
              else{
                oddsum += Arr[i];
              }
            }
            
            A[0] = evensum;

Q11. Birthday Game?

           
Ans. function Birthday_Game(arr,D ,M) {
 
            let count = 0;
            
             for(let i = 0; i < arr.length - 1; i++){
               
                   let subArray = arr.slice(i, i + M);
                   
                   if (subArray.length === M) {
                     
                      let sum = subArray.reduce((a,b) => a + b, 0);
                      
                      if(sum === D){
                        
                           count += 1;
                       }
                    }
              }
              return count;
         
         }
